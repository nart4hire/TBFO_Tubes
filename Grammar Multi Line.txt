S ->  DEF | IMPORT | CLASS | SUITE | WITH | Pass

SUITE -> IF | LOOP | ASSIGNMENT | METHOD | SUITE

ASSIGNMENT -> Var

IF -> If | If Raise | If ELIF Else | If Else | If SUITE ELIF SUITE Else | If SUITE Else | If SUITE ELIF Else | If ELIF SUITE Else

ELIF -> ELIF Elif | Elif

LOOP -> WHILE | FOR

WHILE -> While | While CONTROL | While SUITE CONTROL

FOR -> For | For CONTROL | For SUITE CONTROL

DEF -> Def ISI

ISI -> SUITE Return | ISI ISI | Return

CONTROL -> Break | Continue

Return -> "return"
Def -> "def"
While -> "while"
For -> "for"
Break -> "break"
Continue -> "continue"
If -> "if"
Elif -> "elif"
Else -> "else"
Pass -> "pass"
Raise -> "raise"
